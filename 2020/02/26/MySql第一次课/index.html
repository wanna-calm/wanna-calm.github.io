一、什么是数据库
用来存储数据的仓库，数据库由一张一张的表
组成，真正存储数据的地方为表
二、什么是表
表由行和列组成
行在数据库里称为记录
列称为字段或属性
例如：学生表，表里应该有，学生的学号，
姓名，身份证号，年纪。那么这些就称为
表的字段。
关系型数据库，是由一张一张有关系的表的
组成的
三、安装Mysql
先安装Mysql服务器
数据库是创建的在服务器上的
再安装Mysql客户端
用于连接上mysql服务器，进而对
mysql服务器进行操作，比较常用
客户端有navicat sqlyog
四、sql语句的学习
1、创建语句
1.1、创建数据库

字段名1 数据类型 约束,
字段名2 数据类型 约束,
字段名3 数据类型
)

int
char 字符串类型，固定
长度

长度

4、删除语句

#创建库

CREATE DATABASE onlinemalll;

#使用指定的库

USE onlinemalll;

#创建表

CREATE TABLE goods(
gid INT PRIMARY KEY,#把gid设置
#表的主键
gnam VARCHAR(100),
gprice FLOAT
)

#插入数据

INSERT INTO goods(gid,gnam,gprice)
VALUES(1,’小熊雨伞’,11.6);
INSERT INTO goods(gid,gnam,gprice)
VALUES(2,’小熊饼干’,3.5);
INSERT INTO goods(gid,gnam,gprice)
VALUES(3,’小熊书包’,25);

#查询goods表

SELECT * FROM goods;

#删除gid等于3的数据

DELETE FROM goods WHERE gid=3;

#把小熊雨伞的价格改为15元

UPDATE goods SET gprice=15
WHERE gnam=’小熊雨伞’;
